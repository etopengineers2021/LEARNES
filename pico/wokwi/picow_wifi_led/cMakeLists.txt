# Generated Cmake Pico project file

cmake_minimum_required(VERSION 3.13)

set(PICO_SDK_PATH "C:/Users/ishar/.pico-sdk/sdk/2.2.0")
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# VS Code Pico extension setup
if(WIN32)
    set(USERHOME $ENV{USERPROFILE})
else()
    set(USERHOME $ENV{HOME})
endif()

set(sdkVersion 2.2.0)
set(toolchainVersion 14_2_Rel1)
set(picotoolVersion 2.2.0-a4)
set(picoVscode ${USERHOME}/.pico-sdk/cmake/pico-vscode.cmake)

if (EXISTS ${picoVscode})
    include(${picoVscode})
endif()

# IMPORTANT: Pico W board
set(PICO_BOARD pico_w CACHE STRING "Board type")

# Pull in Raspberry Pi Pico SDK
include(${PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(picow_wifi_led C CXX ASM)

pico_sdk_init()

add_executable(picow_wifi_led
    main.c
)

pico_set_program_name(picow_wifi_led "picow_wifi_led")
pico_set_program_version(picow_wifi_led "0.1")

# Check WiFi name/password
if (NOT DEFINED WIFI_SSID)
    message(FATAL_ERROR "Please pass WIFI_SSID. Example: -DWIFI_SSID=\"YourWiFiName\"")
endif()

if (NOT DEFINED WIFI_PASSWORD)
    message(FATAL_ERROR "Please pass WIFI_PASSWORD. Example: -DWIFI_PASSWORD=\"YourPassword\"")
endif()

target_compile_definitions(picow_wifi_led PRIVATE
    WIFI_SSID=\"${WIFI_SSID}\"
    WIFI_PASSWORD=\"${WIFI_PASSWORD}\"
)

target_include_directories(picow_wifi_led PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}
)

target_link_libraries(picow_wifi_led
    pico_stdlib
    pico_cyw43_arch_lwip_threadsafe_background
)

pico_enable_stdio_usb(picow_wifi_led 1)
pico_enable_stdio_uart(picow_wifi_led 0)

pico_add_extra_outputs(picow_wifi_led)
